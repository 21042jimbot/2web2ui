## Usage

### As an Interactive Element

For elements that trigger page behavior or submit forms.

```js
import { Button } from 'src/components/matchbox';

const MyComponent = () => {
  return (
    <Button variant="primary" onClick={() => console.log('hello, world')}>
      Click Me
    </Button>
  )
}
```

### As a Link

For elements that leverage the app's router or link to an external URL.

```js
import { PageLink } from 'src/components/links';
import { Button } from 'src/components/matchbox';

const MyComponent = () => {
  return (
    <PageLink as={Button} to="https://sparkpost.com">Go to SparkPost.com</PageLink>
  )
}
```
